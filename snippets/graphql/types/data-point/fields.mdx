<ResponseField name="id" type="ID!">
  The unique identifier of the data point.
</ResponseField>
<ResponseField name="environmentId" type="ID!">
  The unique identifier of the environment that the data point belongs to.
</ResponseField>
<ResponseField name="dataBucketId" type="ID!">
  The unique identifier of the data bucket that the data point belongs to.
</ResponseField>
<ResponseField name="dataTypeId" type="ID!">
  The unique identifier of the data type that the data point belongs to.
</ResponseField>
<ResponseField name="attributes" type="JSONObject!">
  An object containing the data point's attributes that can be used to filter
  data points.
</ResponseField>
<ResponseField name="metadata" type="JSONObject!">
  An object containing metadata for the data point.
</ResponseField>
<ResponseField name="value" type="Float!">
  The value of the data point.
</ResponseField>
